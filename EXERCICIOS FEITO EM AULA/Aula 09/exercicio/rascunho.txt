void lerArq(){
	FILE *F;
	F = fopen("cadastroARQ\\arquivo.txt", "r");
	if (F==NULL){
		print("ERRO AO LER O ARQUIVO!!!");
		system("pause");
		exit (0);
	}else{
		print("ARQUIVO CARREGADO!!!\n\n\n");
	}
	char c = fget(F);
	while(c!=EOF){ //enquanto c nao for final de arquivo, ou seja, ir√° ler e imprimir cada caractere
		printf("%c", c);
		c=fget(F);
	}
}

void addrArq(){

		exit(0);
	}
	
	char dados[32]="\n Aula de estrutua de dados, A";
	int i = 0;
	for(i=0; i<strlen(dados); i++){
		fputc(dados[i], F);

}
